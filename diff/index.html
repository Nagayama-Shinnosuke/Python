URLリストを作る

同じフォルダにurls.csvを作成

内容テンプレート


url
https://example.com
https://example.org

Pythonコードを作る

同じフォルダにget_h1.pyファイル作成


import requests
from bs4 import BeautifulSoup
import pandas as pd
import time

# URLリスト読み込み
df = pd.read_csv("urls.csv")

results = []

for url in df["url"]:
    try:
        print("取得中:", url)

        response = requests.get(url, timeout=10)
        soup = BeautifulSoup(response.text, "html.parser")

        h1 = soup.find("h1")

        if h1:
            h1_text = h1.get_text(strip=True)
        else:
            h1_text = ""

        results.append({
            "url": url,
            "h1": h1_text
        })

        time.sleep(1)

    except Exception as e:

        results.append({
            "url": url,
            "h1": "取得失敗"
        })

# CSVとして保存
result_df = pd.DataFrame(results)
result_df.to_csv("result.csv", index=False, encoding="utf-8-sig")

print("完了")

実行する

python get_h1.py
